#include <stdio.h>
#include <string.h>


// 用邻接矩阵存储图


#define maxn 1005
int a[maxn][maxn] ;


//a[x][y] != 0 : 有通路，且长度为a[x][y]
//否则就是没有通路


int n , m ;


//判断它是不是在生成树里
int vis[maxn] ;
int mdis[maxn] ;//点到生成树里的点的最短距离。


#define min(x,y) ((x)<(y)?(x):(y))




int main () {
    //输入点的个数和边的个数
    //接下来每行都会输入一条边两头的点，以及边的长度。


    scanf ("%d%d" , &n , &m) ;


    for (int i = 1 ; i <=m  ; ++ i) {


        int x , y ,  z ;
        scanf ("%d%d%d" , &x , &y , &z) ;
        a[x][y] = z ;
        a[y][x] = z ;


        //连边。
    }


    //stp.1     找到一个距离最小的点
    //stp.2     标记它加入最小生成树
    //stp.3     更新其他点的距离。


    memset (mdis , 0x3f, sizeof (mdis)) ;
    vis[1] = 1 ;
    for (int i = 1 ; i <= n ; ++ i) {
        if (a[1][i]) {


            mdis[i] = min (mdis[i] , a[1][i]) ;
        }
    }


    int ans = 0 ;
    for (int i = 1 ; i < n ; ++ i) {
        //重复n-1次，每次都把一个点插入最小生成树


        int mini = 0 , minv = 0x3fffffff ;
        //mini时最小距离的下标，minv是最小的距离。
        for (int j = 1 ; j <= n ; ++ j) {
            if (vis[j] == 0) {
                if (mdis[j] < minv) {
                    minv = mdis[j] ;
                    mini = j ;
                }
            }
        }


        ans += minv ;


        //找到还未加入但是距离最短的点。
        vis[mini] = 1 ;
        for (int j = 1 ; j <= n ; ++ j) {
            if (a[mini][j]) {


                mdis[j] = min (mdis[j] , a[mini][j]) ;
            }
        }
    }


    printf ("%d\n" , ans) ;
    return 0 ;
}

#define maxn 20


int q[maxn] ,vis[maxn] ;
int head = 1 , tail = 0;
//对于topsort 来说，所有元素都进一次，出一次。 
int isEmpty () {
    return head > tail ;//这样队列就是空的 
}
int front () {
    return q[head] ;
    //不检查对列本来是不是空的 
}
void push (int x ) {
    
    ++ tail ;
    q[tail] = x ;
}
void pop () {
    ++ head ;
}


bool TopSort( LGraph Graph, Vertex TopOrder[] ) {
    
//插入入度为零的节点。


    for (int i = 0 ; i < Graph->Nv ; ++ i) {
        vis[i] = 0 ;
    }
    
    for (int i = 0 ; i < Graph->Nv ; ++ i) 
    {
        for (PtrToAdjVNode e = Graph->G[i].FirstEdge ; e != NULL ; e = e->Next) {
            vis[e->AdjV] ++ ;
        }
    }
    
    for (int i = 0 ; i < Graph->Nv ; ++ i) {
        if (vis[i] == 0) {
            push (i) ;
        }
    }
    
    int cnt = -1 ;
    while (!isEmpty ()) {
        
        int x = front () ;
        pop () ;
        
        //输出拓扑序列
        ++ cnt ;
        TopOrder[cnt] = x ; 
        //从队首拿出一个元素来
        
        for (PtrToAdjVNode e = Graph->G[x].FirstEdge ; e != NULL ; e = e->Next) {
            
            vis[e->AdjV] -- ;
            //把边的另一端减少 
            
            if (vis[e->AdjV] == 0) {
                push (e->AdjV) ;
            }
        } 
        
        
        
    }


    if (cnt == Graph->Nv) return true ;
    else return false ;
}

#include <stdio.h>
#include <stdlib.h>


//用来排序的
//用来比较两个数/复杂的结构体哪一个大或者小。


//想要把一群点按照x从小到大，如果x相同，就按照y从小到大排序。
struct Point {


    int x , y ;
};
int cmp (const void *pa , const void *pb) {


    struct Point * a = (struct Point *) pa ;
    struct Point * b = (struct Point *) pb ;


    if (a->x != b->x) {


        if (a->x < b->x) return -1 ;
        else return 1;
    } else {
        if (a->y < b->y) return -1 ;
        else return 1 ;
    }
}


struct Point a[100] ;
int main () {


    srand (2) ;
    int n = 10 ;
    for (int i = 1 ; i <= n ; ++ i ) {
        a[i].x = rand () % 10, a[i].y = rand () % 10;
//        printf ("%d %d\n" , a[i].x , a[i].y) ;
    }


    //提供一个首指针，提供一个尾指针的后一位。
    qsort (a + 1, n , sizeof (struct Point) ,cmp ) ;

//        a[i].x = rand () , a[i].y = rand () ;
        printf ("%d %d\n" , a[i].x , a[i].y) ;

}

#include <stdio.h>


//数组实现的栈


int t = 0 ;
//t代表的是栈里面的元素的数量
//t还有一个任务就是，栈顶的元素的数组下标
 
#define maxn 1005
int s[maxn] ;


//判断栈是否为空
int isEmpty () {
    return t == 0 ;
}


int top () {
    //判断栈是否为空就省略了 
    return s[t] ;
}


void pop () {
    //删除栈顶元素。 
    -- t ;
} 


void push (int x ) {
    s[++ t] = x ;
}
int main () {
    
    push (1) ;
    
    push (2) ;
    
    pop () ;
    printf ("%d\n" , top()) ;
    return 0 ;
}

#include <stdio.h>


//数组模拟的队列！！！


#define maxn 1005
int head = 1 , tail = 0 ;
//分别是首和尾的指针。 
int q[maxn] ; 


int isEmpty () {
    return head > tail ;
} 


int front () {
    //返回首个元素 
    return q[head] ;
}


void pop () {
    //依然没有判断是否为空。 
    head ++ ;
}


void push (int x ) {
    q[++ tail] = x ;
}
int main () {
    
    printf ("%d\n" , isEmpty ()) ;
    
    push(1) ;
    push(2) ;
    
    printf ("%d\n" , front()) ;
    pop () ;
    printf ("%d\n" , front()) ;
    return 0 ;
}
