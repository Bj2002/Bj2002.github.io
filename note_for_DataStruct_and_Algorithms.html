计算最大公约数

  • 辗转相除法

    • 设两数分别为a和b。求a和b的最大公因数gcd(a,b)
    • 不妨设a>b，如果发现b是a的因子，那么b就是二者的最大公约数。如果不能整除，则a=bq+r，其中r<b
    • 即gcd(a,b)=gcd(b,amodb)
    • 参考代码
int gcd (int x, int y) {
    if (y == 0) return x ;
    return gcd (y, x % y) ;
}

 

  • 质因数分解：对两束进行质因数分解，然后取相同的部分。

// C++ version
#include <bits/stdc++.h>
using namespace std ;


#define MAX_PRIME 10005
int p[MAX_PRIME] ;
vector <int>primeList ;




//获取素数因子列表，存储在res里
void getPrimeFactor (int x, vector <int >&res) {
    res.clear() ;


    for (int i = 0 ; i < primeList.size() && primeList[i] <= x; ++ i) {


        while (x % primeList[i] == 0) {
            res.push_back(primeList[i]) ;
            x /= primeList[i] ;
        }
    }
}


int gcd (int x, int y) {
    vector <int> factorList1, factorList2 ;
    getPrimeFactor(x, factorList1) ;
    getPrimeFactor(y, factorList2) ;


    int i = 0, j = 0 ;


    int ans = 1 ;
    while (i < factorList1.size() && j < factorList2.size()) {


        while (factorList1[i] < factorList2[j] && i < factorList1.size()) ++ i ;
        while (factorList1[i] > factorList2[j] && j < factorList2.size()) ++ j ;


        if (factorList1[i] == factorList2[j]) {


            //            printf ("%d %d %d\n", i, j, factorList1[i]) ;
            ans *= factorList1[i] ;
            ++ i ;
            ++ j ;
        }
    }
    return ans ;
}




int main () {


    memset (p, -1, sizeof (p)) ;


    //筛选素数，存储在primeList里
    p[1] = 0 ;
    for (int i = 1; i < MAX_PRIME ; ++ i) {
        if (p[i] != 0) {
            for (int j = i * i ; j < MAX_PRIME ; j += i) {
                p[j] = 0 ;
            }


            primeList.push_back(i) ;
        }
    }


    int a, b;
    scanf ("%d%d", &a, &b) ;
    printf ("%d\n", gcd (a, b)) ;
    return 0 ;
}
